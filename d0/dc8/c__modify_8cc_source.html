00036 /* faultbranch: "fault" a single branch. (temporarily change a value)
00037  * save the existing info in "faultlist", then patch
00038  */
00039 void faultbranch(CARD* brh, double value)
00040 {
00041   if (!brh->is_device()) {
00042     untested();

00056 /* sweep_fix: fix the value for sweep command.
00057  * (find value by interpolation)
00058  * if not sweeping, return "start" (the first arg).
00059  */
00060 double sweep_fix(CS& cmd, const CARD *brh)
00061 {
00062   double start = cmd.ctof();
00063   double value = start;
00064   if (swp_steps[swp_nest] != 0   &&   cmd.is_float()) {
00065     untested();
00066     double last = cmd.ctof();
00067     double offset = static_cast<double>(swp_count[swp_nest]) 
00068       / static_cast<double>(swp_steps[swp_nest]);
00069     if (swp_type[swp_nest]=='L') {
00070       untested();
00071       if (start == 0.) {
00072         untested();
00073         throw Exception("log sweep can't pass zero");
00074         value = 0;
00075       }else{
00076         untested();
00077         value = start * pow( (last/start), offset );
00078       }
00079     }else{
00080       untested();
00081       value = start + (last-start) * offset;
00082     }
00083     IO::mstdout.setfloatwidth(7)
00084       << swp_count[swp_nest]+1 << "> sweep " << brh->long_label()
00085       << " =" << value << '\n';
00086   }
00087   return value;
00088 }

00090 void modify_fault(CS& cmd, WHATTODO command, CARD_LIST* scope)
00091 {
00092   CKT_BASE::_sim->uninit();
00093   while (cmd.is_alpha()) {
00094     unsigned mark = cmd.cursor();
00095     unsigned cmax = cmd.cursor();
00096     CARD_LIST::fat_iterator ci(scope, scope->begin());
00097     for (;;) {
00098       cmd.reset(mark);
00099       ci = findbranch(cmd, ci);
00100       cmax = std::max(cmax, cmd.cursor());
00101       if (ci.is_end()) {
00102         break;
00103       }
00104       cmd.skip1b('=');
00105       CARD* brh = *ci;
00106       switch (command) {
00107       case MODIFY:{
00108         ELEMENT* e = prechecked_cast<ELEMENT*>(brh);
00109         assert(e);
00110         e->set_value(cmd.ctof(), 0);
00111         //BUG// messy way to do this, loses other attributes
00112       } break;
00113       case FAULT:{
00114         faultbranch(brh, sweep_fix(cmd,brh));
00115       } break;
00116       }
00117       cmax = std::max(cmax, cmd.cursor());
00118       ++ci;
00119     }
00120     cmd.reset(cmax);
00121     if (mark == cmax) {
00122       untested();
00123       cmd.check(bWARNING, "what's this?");
00124       cmd.skiparg();
00125     }
00126   }
00127 }
