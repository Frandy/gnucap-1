00100 void EVAL_BM_TANH::tr_eval(ELEMENT* d)const
00101 {
00102   double x = ioffset(d->_y[0].x);
00103   double aa = x * _gain/_limit;
00104   double f1, f0;
00105   if (aa > LOGBIGBIG) {
00106     f1 = 0;
00107     f0 = _limit;
00108   }else if (aa < -LOGBIGBIG) {
00109     f1 = 0;
00110     f0 = -_limit;
00111   }else{
00112     double cosine = cosh(aa);
00113     f1 = _gain / (cosine*cosine);
00114     f0 = _limit * tanh(aa);
00115   }
00116   d->_y[0] = FPOLY1(x, f0, f1);
00117   tr_final_adjust(&(d->_y[0]), d->f_is_value());
00118 }
