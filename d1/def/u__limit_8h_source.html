00029 /* Spice style PN junction limiting */
00030 inline double pnj_limit(double vnew, double vold, double vt, double vcrit)
00031 {
00032   trace4("limit-in", vnew, vold, vt, vcrit);
00033   if ((vnew > vcrit) && (std::abs(vnew - vold) > (2 * vt))) {
00034     if (vold > 0) {
00035       double arg = 1 + (vnew - vold) / vt;
00036       double vlim = (arg > 0) ? (vold + vt * log(arg)) : vcrit;
00037       trace3("limit-1", vlim, vt*log(arg), arg);
00038       return vlim;
00039     }else{
00040       double vlim = vt * log(vnew/vt);
00041       trace3("limit-2", vlim, vt*log(vnew/vt), vnew/vt);
00042       return vlim;
00043     }
00044   }else{
00045     trace1("limit-3", vnew);
00046     return vnew;
00047   }
00048 }
