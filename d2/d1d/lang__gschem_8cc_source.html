00243 /* find_type_in_string : specific to each language which gives the type of the
00244  * cmd Eg: verilog : comment ,resistor etc
00245  * In Gschem the broad high level types are
00246  * The following don't mean anything to the electrical part of the circuit and
00247  * they are of type graphical
00248  *  version  v    * circle   V
00249  *  line     L    * arc      A
00250  *  picture  G    * path     H
00251  *  box      B    * text     T (not attributes)
00252  * The following have electrical meaning and are of type
00253  *  net          N
00254  *  bus          U
00255  *  pin          P
00256  *  component    C
00257  *  attributes   T (enclosed in {})  // don't include in find_type_in_string
00258  * Will check for type graphical (type=dev_comment) else type will be
00259  * net or bus or pin or component\
00260  */
00261 std::string LANG_GSCHEM::find_type_in_string(CS& cmd)
00262 {
00263     unsigned here = cmd.cursor(); //store cursor position to reset back later
00264     std::string type;   //stores type : should check device attribute..
00265     if (cmd >> "v" || cmd >> "L" || cmd >> "G" || cmd >> "B" || cmd >>"V" ||
00266         cmd >> "A" || cmd >> "H" || cmd >> "T"){ type="dev_comment";}
00267     else if (cmd >> "N"){ type="net";}
00268     else if (cmd >> "U"){ type="bus";}
00269     else if (cmd >> "P"){ type="pin";}
00270     else if (cmd >> "C"){ type="C";}
00271     else {  cmd >> type; } //Not matched with the type. What now?
00272     cmd.reset(here);//Reset cursor back to the position that
00273                     //has been started with at beginning
00274     return type;    // returns the type of the string
