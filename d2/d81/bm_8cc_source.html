00091 void EVAL_BM_ACTION_BASE::ac_final_adjust(COMPLEX* y)const
00092 {
00093   if (_bandwidth != NOT_INPUT && _bandwidth != 0.) {untested();
00094     assert(y->imag() == 0);
00095     double ratio = CKT_BASE::_sim->_freq / _bandwidth;
00096     double coeff = y->real() / (1.+(ratio*ratio));
00097     *y = COMPLEX(coeff, -coeff * ratio);
00098   }else{
00099   }
00100   
00101   if (_phase != 0.) {itested();
00102     *y *= std::polar(1., _phase*DTOR);
00103   }else{
00104   }
00105 
00106   if (_delay != 0.) {untested();
00107     double ratio = CKT_BASE::_sim->_freq * _delay;
00108     if (ratio > 100000.) {untested();
00109       error(bPICKY, "delay too long\n");
00110       ratio = 0.;
00111     }else{untested();
00112     }
00113     *y *= std::polar(1., -360.0 * DTOR * ratio);
00114   }else{
00115   }
00116 }
