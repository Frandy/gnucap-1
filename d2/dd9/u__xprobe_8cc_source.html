00029 double XPROBE::operator()(mod_t m, bool db)const
00030 {
00031   if (exists()) {
00032     if (m == mtNONE) {
00033       m = _modifier;
00034     }
00035     double rv = NOT_VALID;
00036     switch (m) {
00037     case mtNONE:
00038       unreachable();
00039       break;
00040     case mtMAG:
00041       rv = std::abs(_value);
00042       break;
00043     case mtPHASE:
00044       rv = phase(_value);
00045       break;
00046     case mtREAL:
00047       rv = real(_value);
00048       break;
00049     case mtIMAG:
00050       rv = imag(_value);
00051       break;
00052     }
00053     return (db) ? _dbscale * log10(std::max(rv,VOLTMIN)) : rv;
00054   }else{
00055     return NOT_VALID;
00056   }
00057 }
