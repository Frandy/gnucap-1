00029 /* skipbl: skip whitespace.  (any non-graphic character is ws)
00030  * =,(,) are also ws
00031  * update string pointer
00032  * pointer points to first non-space
00033  * does NOT update _ok flag
00034  */
00035 CS& CS::skipbl()
00036 {
00037   while (peek() && (!isgraph(peek()))) {
00038     skip();
00039   }
00040   return *this;
00041 }
