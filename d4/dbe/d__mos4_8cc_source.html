00815 
00816     trace3("", d->vds, d->vgs, d->vbs);

00825       untested();

00829       dUgsdVbs = s->ugsB;
00830     }

00832     
00833     double Uds = s->uds + s->udsB * d->vbs + s->udsD * (d->vds - m->vdd);
00834     double dUdsdVbs;
00835     double dUdsdVds;
00836     if(Uds <= 0) {    
00837       untested();

00844       dUdsdVds = s->udsD / Leff;

00859     /* threshold voltage */
00860     double dVthdVbs;
00861     double dVthdVds;

00864       double dEtadVds;

00876         dEtadVbs = s->etaB;
