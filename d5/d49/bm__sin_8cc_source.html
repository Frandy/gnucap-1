00150 void EVAL_BM_SIN::tr_eval(ELEMENT* d)const
00151 {
00152   if (!_frequency.has_hard_value()) {
00153     _actual_frequency = d->_sim->_freq;
00154   }else{
00155     _actual_frequency = _frequency;
00156   }
00157 
00158   double reltime = ioffset(d->_sim->_time0);
00159   double ev = _offset;
00160   if (reltime > _delay) {
00161     double x=_amplitude*fixzero(sin(M_TWO_PI*_actual_frequency*(reltime-_delay)),1.);
00162     if (_damping != 0.) {
00163       x *= exp(-(reltime-_delay)*_damping);
00164     }else{
00165     }
00166     ev += x;
00167   }else{
00168   }
00169   tr_finish_tdv(d, ev);
00170 }

00172 TIME_PAIR EVAL_BM_SIN::tr_review(COMPONENT* d)
00173 {
00174   double reltime = ioffset(d->_sim->_time0) + d->_sim->_dtmin * .01;
00175 
00176   if (reltime > _delay) {
00177     if (_peak && _zero) {
00178       d->_time_by.min_event(floor(reltime * 4 * _actual_frequency + 1) / (4 * _actual_frequency));
00179     }else if (_peak) {
00180       d->_time_by.min_event((floor(reltime*2*_actual_frequency+.5) + .5) / (2*_actual_frequency));
00181     }else if (_zero) {
00182       d->_time_by.min_event(floor(reltime * 2 * _actual_frequency + 1) / (2 * _actual_frequency));
00183     }else{
00184     }
00185     d->_time_by.min_error_estimate(d->_sim->_time0 + 1. / (_samples * _actual_frequency));
00186   }else{
00187     d->_time_by.min_event(_delay);
00188   }
00189 
00190   return d->_time_by;
00191 }
