00122 /* add_list: add a "list" of probes, usually only one
00123  * This means possibly several probes with a single parameter
00124  * like "v(r*)" meaning all resistors
00125  * but not "v(r4) v(r5)" which has two parameters.
00126  * It also takes care of setting the range for plot or alarm.
00127  */
00128 void PROBELIST::add_list(CS& cmd)
00129 {
00130   int oldcount = size();
00131   std::string what(cmd.ctos(TOKENTERM));/* parameter */
00132   if (what.empty()) {untested();
00133     cmd.warn(bWARNING, "need a probe");
00134   }else{
00135   }
00136 
00137   int paren = cmd.skip1b('(');          /* device, node, etc. */
00138   if (cmd.umatch("nodes ")) {
00139     // all nodes
00140     add_all_nodes(what);
00141   }else if (cmd.umatch("0")) {
00142     // node 0 means system stuff
00143     push_new_probe(what, 0);
00144   }else if (cmd.is_alnum() || cmd.match1("*?")) {
00145     // branches or named nodes
00146     unsigned here1 = cmd.cursor();
00147     bool found_something = add_branches(cmd.ctos(),what,&CARD_LIST::card_list);
00148     if (!found_something) {
00149       cmd.warn(bWARNING, here1, "no match");
00150     }else{
00151     }
00152     for (;;) {
00153       // a list, as in v(r1,r2,r3) or v(1,2,3)
00154       if (!(cmd.is_alnum() || cmd.match1("*?"))) {
00155         break;
00156       }else{
00157       }
00158       unsigned here2 = cmd.cursor();
00159       found_something = add_branches(cmd.ctos(),what,&CARD_LIST::card_list);
00160       if (!found_something) {itested();
00161         cmd.reset(here2);
00162         break;
00163       }else{
00164       }
00165     }
00166   }else{itested();
00167     cmd.warn(bDANGER, "need device or node");
00168   }
00169   paren -= cmd.skip1b(')');
00170   if (paren != 0) {itested();
00171     cmd.warn(bWARNING, "need )");
00172   }else{
00173   }
00174 
00175   if (cmd.skip1b('(')) {        /* range for plotting and alarm */
00176     double lo = cmd.ctof();
00177     double hi = cmd.ctof();

00179       p->set_limit(lo,hi);

00184     }
00185   }else{
00186   }
