00081 /* map__nodes: map intermediate node number to internal node number.
00082  * Ideally, this function would find some near-optimal order
00083  * and squash out gaps.
00084  */
00085 void SIM_DATA::map__nodes()
00086 {
00087   _nm = new int[_total_nodes+1];
00088   ::status.order.reset().start();
00089   switch (OPT::order) {
00090     default: untested();
00091       error(bWARNING, "invalid order spec: %d\n", OPT::order);
00092     case oAUTO:                order_auto();    break;
00093     case oREVERSE: untested(); order_reverse(); break;
00094     case oFORWARD: untested(); order_forward(); break;
00095   }
00096   ::status.order.stop();
00097 }
