00034     PARAMETER<double> before(BIGBIG);
00035     PARAMETER<double> after(-BIGBIG);
00036     
00037     unsigned here = Cmd.cursor();
00038     Cmd >> probe_name;
00039     WAVE* w = find_wave(probe_name);
00040 
00041     if (!w) {
00042       Cmd.reset(here);
00043     }else{
00044     }
00045 
00046     here = Cmd.cursor();
00047     do {
00048       ONE_OF
00049         || Get(Cmd, "probe",  &probe_name)
00050         || Get(Cmd, "before", &before)
00051         || Get(Cmd, "after",  &after)
00052         || Get(Cmd, "end",    &before)
00053         || Get(Cmd, "begin",  &after)
00054         ;
00055     }while (Cmd.more() && !Cmd.stuck(&here));
00056 
00057     if (!w) {
00058       w = find_wave(probe_name);
00059     }else{
00060     }
00061 
00062     if (w) {
00063       before.e_val(BIGBIG, Scope);
00064       after.e_val(-BIGBIG, Scope);
00065 
00066       WAVE::const_iterator begin = lower_bound(w->begin(), w->end(), DPAIR(after, -BIGBIG));
00067       WAVE::const_iterator end   = upper_bound(w->begin(), w->end(), DPAIR(before, BIGBIG));
00068       WAVE::const_iterator lower = begin;
00069 
00070       double area = 0;
00071       WAVE::const_iterator i = begin;
00072       while (++i < end) {
00073         area += .5 * (lower->second*lower->second + i->second*i->second)
00074           * (i->first - lower->first);
00075         lower = i;
00076       }
00077       return to_string(sqrt(area/(lower->first - begin->first)));
00078     }else{
00079       throw Exception_No_Match(probe_name);
00080     }
00081   }
00082 } p4;
