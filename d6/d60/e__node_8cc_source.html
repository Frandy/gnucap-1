00172 double NODE::tr_probe_num(const std::string& x)const
00173 {
00174   if (Umatch(x, "v ")) {
00175     // return v0(); denoised
00176     return floor(v0()/OPT::vfloor + .5) * OPT::vfloor;
00177   }else if (Umatch(x, "z ")) {
00178     return port_impedance(node_t(const_cast<NODE*>(this)), node_t(&ground_node), _sim->_lu, 0.);
00179   }else if (Umatch(x, "l{ogic} |la{stchange} |fi{naltime} |di{ter} |ai{ter} |count ")) {
00180     assert(_sim->_nstat);
00181     return _sim->_nstat[matrix_number()].tr_probe_num(x);
00182   }else if (Umatch(x, "mdy ")) {
00183     // matrix diagonal admittance
00184     const BSMATRIX<double>&  aaa = _sim->_aa;
00185     return aaa.d(m_(),m_());
00186   }else if (Umatch(x, "mdz ")) {
00187     // matrix diagonal impedance
00188     const BSMATRIX<double>&  aaa = _sim->_aa;
00189     return 1/aaa.d(m_(),m_());
00190   }else if (Umatch(x, "zero ")) {
00191     // fake probe: 0.0
00192     return 0.0;
00193   }else if (Umatch(x, "pdz ")) {
00194     // fake probe 1/0 .. positive divide by zero = Infinity
00195     double z1 = tr_probe_num("zero ");
00196     return 1.0/z1;
00197   }else if (Umatch(x, "ndz ")) {
00198     // fake probe -1/0 .. negative divide by zero = -Infinity
00199     double z1 = tr_probe_num("zero ");
00200     return -1.0/z1;
00201   }else if (Umatch(x, "nan ")) {
00202     // fake probe 0/0 = NaN
00203     double z1 = tr_probe_num("zero ");
00204     double z2 = tr_probe_num("zero ");
00205     return z1/z2;
00206   }else{itested();
00207     return CKT_BASE::tr_probe_num(x);
00208   }
00209 }

00275   }

00277

00279     if (_sim->analysis_is_restore()) {untested();
00280     }else if (_sim->analysis_is_static()) {
00281     }else{
00282     }
00283     if (_sim->analysis_is_static() || _sim->analysis_is_restore()) {
00284       set_last_change_time(0);
00285       store_old_last_change_time();
00286       set_lv(lvUNKNOWN);
00287     }else{
00288     }
00289     double dt = _sim->_time0 - last_change_time();
00290     if (dt < 0.) {untested();
00291       error(bPICKY, "time moving backwards.  was %g, now %g\n",
00292             last_change_time(), _sim->_time0);
00293       dt = _sim->_time0 - old_last_change_time();
00294       if (dt <= 0.) {untested();
00295         throw Exception("internal error: time moving backwards, can't recover");
00296       }else{untested();
00297       }
00298       assert(dt > 0.);
00299       restore_lv();                     /* skip back one */
00300     }else{
00301       store_old_last_change_time();
00302       store_old_lv();                   /* save to see if it changes */
00303     }
00304     
00305     double sv = v0() / process()->range;        /* new scaled voltage */
00306     if (sv >= process()->th1) {         /* logic 1 */
00307       switch (lv()) {
00308       case lvSTABLE0: dont_set_quality("stable 0 to stable 1"); break;
00309       case lvRISING:  dont_set_quality("begin stable 1");       break;
00310       case lvFALLING:untested();set_bad_quality("falling to stable 1"); break;
00311       case lvSTABLE1: dont_set_quality("continuing stable 1");  break;
00312       case lvUNKNOWN: set_good_quality("initial 1");            break;
00313       }
00314       set_lv(lvSTABLE1);
00315     }else if (sv <= process()->th0) {   /* logic 0 */
00316       switch (lv()) {
00317       case lvSTABLE0: dont_set_quality("continuing stable 0");  break;
00318       case lvRISING: untested();set_bad_quality("rising to stable 0");  break;
00319       case lvFALLING: dont_set_quality("begin stable 0");       break;
00320       case lvSTABLE1: dont_set_quality("stable 1 to stable 0"); break;
00321       case lvUNKNOWN: set_good_quality("initial 0");            break;
00322       }
00323       set_lv(lvSTABLE0);
00324     }else{                              /* transition region */
00325       double oldsv = vt1() / process()->range;/* old scaled voltage */
00326       double diff  = sv - oldsv;
00327       if (diff > 0) {   /* rising */
00328         switch (lv()) {
00329         case lvSTABLE0:
00330           dont_set_quality("begin good rise");
00331           break;
00332         case lvRISING:
00333           if (diff < dt/(process()->mr * process()->rise)) {
00334             set_bad_quality("slow rise");
00335           }else{
00336             dont_set_quality("continuing good rise");
00337           }
00338           break;
00339         case lvFALLING:
00340           untested();
00341           set_bad_quality("positive glitch in fall");
00342           break;
00343         case lvSTABLE1:
00344           untested();
00345           set_bad_quality("negative glitch in 1");
00346           break;
00347         case lvUNKNOWN:
00348           set_bad_quality("initial rise");
00349           break;
00350         }
00351         set_lv(lvRISING);
00352       }else if (diff < 0) {     /* falling */
00353         switch (lv()) {
00354         case lvSTABLE0:
00355           untested();
00356           set_bad_quality("positive glitch in 0");
00357           break;
00358         case lvRISING:
00359           set_bad_quality("negative glitch in rise");
00360           break;
00361         case lvFALLING:
00362           if (-diff < dt/(process()->mf * process()->fall)) {
00363             set_bad_quality("slow fall");
00364           }else{
00365             dont_set_quality("continuing good fall");
00366           }
00367           break;
00368         case lvSTABLE1:
00369           dont_set_quality("begin good fall");
00370           break;
00371         case lvUNKNOWN:
00372           untested();
00373           set_bad_quality("initial fall");
00374           break;
00375         }
00376         set_lv(lvFALLING);
00377       }else{                            /* hanging up in transition */
00378         untested();
00379         error(bDANGER, "inflection???\n");
00380         set_bad_quality("in transition but no change");
00381         /* state (rise/fall)  unchanged */
00382       }
00383     }
00384     if (sv > 1.+process()->over || sv < -process()->over) {/* out of range */
00385       set_bad_quality("out of range");
00386     }
00387     if (just_reached_stable()) { /* A bad node gets a little better */
00388       improve_quality();        /* on every good transition.       */
00389     }                           /* Eventually, it is good enough.  */
00390                                 /* A good transition is defined as */
00391                                 /* entering a stable state from    */
00392                                 /* a transition state.             */

00424     risefall = process()->fall;
00425     break;
00426   case lvSTABLE1:

00432   assert(end   != NOT_VALID);
00433   assert(risefall != NOT_VALID);

00435   if (_sim->_time0 <= (final_time()-risefall)) {
00436     return start;
00437   }else if (_sim->_time0 >= final_time()) {
00438     untested();
00439     return end;
00440   }else{
00441     untested();
00442     return end - ((end-start) * (final_time()-_sim->_time0) / risefall);
