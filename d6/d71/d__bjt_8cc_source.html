00751 
00752     // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
00753     trace0("--------------------------");
00754     trace1(d->long_label().c_str(), d->evaliter());
00755     trace4("", d->vbe, d->vbc, d->vbx, d->vcs);
00756     // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
00757     double cbe, gbe;
00758     { // d->cpi, d->gpi, d->cpixxx
00759       // uses: d->vbe
00760       double cben, gben;
00761       double vtn = t->vt * m->nf;
00762       double csat = t->ibe * c->area;
00763       double C2 = t->BEleakCur * c->area;
00764       if (d->vbe > -5 * vtn) {
00765         double evbe = exp(d->vbe / vtn);
00766         cbe = csat * (evbe-1) + OPT::gmin * d->vbe;
00767         gbe = csat * evbe/vtn + OPT::gmin;
00768         if (C2 == 0.) {
00769           cben = 0.;
00770           gben = 0.;
00771         }else{
00772           double vte = m->ne * t->vt;
00773           double evben = exp(d->vbe / vte);
00774           cben = C2 * (evben-1);
00775           gben = C2 * evben/vte;
00776         }
00777         trace4("vbe on", cbe, gbe, cben, gben);
00778       }else{
00779         gbe = -csat/d->vbe + OPT::gmin;
00780         cbe = gbe * d->vbe;
00781         gben = -C2 / d->vbe;
00782         cben = gben * d->vbe;
00783         trace4("vbe off", cbe, gbe, cben, gben);
00784       }
00785       d->cpi = cbe / t->BetaF + cben;
00786       d->gpi = gbe / t->BetaF + gben;

00788       trace3("", t->BetaF, d->cpi, d->gpi);
00789     }
00790     // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
00791     double cbc, gbc, cbcn;
00792     { // d->cmu, d->gmu, d->cmuxxx
00793       // uses: d->vbc
00794       double gbcn;
00795       double vtn = t->vt * m->nr;

00797       double C4 = t->BCleakCur * c->area;
00798       if (d->vbc > -5 * vtn) {
00799         double evbc = exp(d->vbc / vtn);
00800         cbc = csat * (evbc-1) + OPT::gmin * d->vbc;
00801         gbc = csat * evbc/vtn + OPT::gmin;
00802         if (C4 == 0.) {
00803           cbcn = 0.;
00804           gbcn = 0.;
00805         }else{
00806           double vtc = m->nc * t->vt;

00808           cbcn = C4 * (evbcn-1);
00809           gbcn = C4 * evbcn/vtc;
00810         }
00811         trace4("vbc on", cbc, gbc, cbcn, gbcn);
00812       }else{
00813         gbc = -csat/d->vbc + OPT::gmin;
00814         cbc = gbc * d->vbc;
00815         gbcn = -C4 / d->vbc;
00816         cbcn = gbcn * d->vbc;
00817         trace4("vbc off", cbc, gbc, cbcn, gbcn);
00818       }
00819       d->cmu = cbc / t->BetaR + cbcn;
00820       d->gmu = gbc / t->BetaR + gbcn;
00821       d->cmuxxx = d->cmu - d->vbc * d->gmu;
00822       trace3("", t->BetaR, d->cmu, d->gmu);
00823     }
00824     // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
00825     //   determine base charge terms
00826     double qb, dqbdve, dqbdvc;
00827     {
00828       double q1 = 1 / (1 - m->invearlyvoltf*d->vbc - m->invearlyvoltr*d->vbe);
00829       if(c->oik == 0. && c->oikr == 0.) {
00830         qb = q1;
00831         dqbdve = q1 * qb * m->invearlyvoltr;
00832         dqbdvc = q1 * qb * m->invearlyvoltf;
00833         trace4("!oik", q1, qb, dqbdve, dqbdvc);
00834       }else{
00835         double q2 = c->oik * cbe + c->oikr * cbc;
00836         double arg = std::max(0., 1+4*q2);
00837         double sqarg = (arg == 0.) ? 1 : sqrt(arg);
00838         qb = q1 * (1+sqarg) / 2;
00839         dqbdve = q1 * (qb * m->invearlyvoltr + c->oik  * gbe / sqarg);
00840         dqbdvc = q1 * (qb * m->invearlyvoltf + c->oikr * gbc / sqarg);
00841         trace2("", c->oik, c->oikr);
00842         trace4("oik", q1, qb, dqbdve, dqbdvc);
00843       }
00844     }
00845     // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
00846     // weil's approx. for excess phase applied with backward-euler integration
00847     {
00848       double cc = 0.;
00849       double cex = cbe;
00850       double gex = gbe;
00851       if (0 && m->excessphasefactor != 0.) {
00852         unreachable();
00853         incomplete(); // doesn't save old values of cexbc, so disabled
00854         double arg1 = d->_dt_0 / m->excessphasefactor;
00855         double arg2 = 3 * arg1;
00856         arg1 *= arg2;
00857         double denom = 1 + arg1 + arg2;
00858         double arg3 = arg1 / denom;
00859         if (_sim->is_initial_step()) {
00860           d->cexbc_2 = d->cexbc_1 = cbe / qb;
00861         }else{
00862         }
00863         cc = (d->cexbc_1 * (1 + d->_dt_0/d->_dt_1 + arg2)
00864               - d->cexbc_2 * d->_dt_0/d->_dt_1) / denom;
00865         cex *= arg3;
00866         gex *= arg3;
00867       }
00868       d->cexbc_0 = cc + cex / qb;
00869 
00870       d->cce = cc + (cex-cbc)/qb - cbc/t->BetaR - cbcn;
00871       d->go = (gbc + (cex-cbc)*dqbdvc / qb) / qb;
00872       d->gm = (gex - (cex-cbc)*dqbdve / qb) / qb - d->go;
00873       d->ccexxx = d->cce - ((d->vbe - d->vbc) * d->go + d->vbe * d->gm);
00874       trace4("", d->cce, d->go, d->gm, d->cce/t->vt);
00875     }
00876     // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
00877     // d->gx
00878     // should be moved to a private eval
00879     // may use d->cpi, d->cmu, qb
00880     {
00881       if (!OPT::rstray || (!has_nz_value(m->rb) && !has_nz_value(m->rbm))) {
00882         trace3("", m->rb, m->irb, d->gx);
00883         assert(d->gx == NA);
00884       }else{
00885         double rx = NA;
00886         double rbpr = m->rbm / c->area;
00887         double rbpi = m->rb / c->area - rbpr;
00888         if (has_nz_value(m->irb)) {itested();//554
00889           // base resistance lowering at high current
00890           double cb = d->cpi + d->cmu;
00891           double xjrb = m->irb * c->area;
00892           double arg1 = std::max(cb/xjrb, 1e-9);

00894           arg1 = tan(arg2);

00896         }else{
00897           rx = rbpr + rbpi / qb;

00900         assert(rx != NA);
00901         assert(rx != 0.);
00902         d->gx = 1 / rx;

00904       }
00905     }
00906     // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -

00927         }else{
