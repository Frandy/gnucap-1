00028 /* phase: extract phase (degrees) from COMPLEX number
00029  * rotates 90 degrees!  (ref to sine instead of cosine)
00030  */
00031 inline double phase(COMPLEX x)
00032 {
00033   double rv = NOT_VALID;
00034   switch (OPT::phase) {
00035   case pDEGREES:
00036     rv = arg(x)*RTOD;
00037     break;
00038   case pP_DEGREES:untested();
00039     rv = arg(x)*RTOD;
00040     if (rv < 0) {untested();
00041       rv += 360;
00042     }else{untested();
00043     }
00044     break;
00045   case pN_DEGREES:itested();
00046     rv = arg(x)*RTOD;
00047     if (rv > 0) {itested();
00048       rv -= 360;
00049     }else{itested();
00050     }
00051     break;
00052   case pRADIANS:
00053     rv = arg(x);
00054     break;
00055   };
00056   return rv;
00057 }
