00267       _step[Nest] = 2.;

00271     break;
00272   case TIMES:untested();
00273     if (_step[Nest] == 0.  &&  _start[Nest] != 0.) {untested();
00274       _step[Nest] = _stop[Nest] / _start[Nest];
00275     }else{untested();
00276     }
00277     _linswp[Nest] = false;
00278     break;
00279   case OCTAVE:
00280     if (_step[Nest] == 0.) {untested();
00281       _step[Nest] = 1.;
00282     }else{
00283     }
00284     _step[Nest] = pow(2.00000001, 1./_step[Nest]);
00285     _linswp[Nest] = false;
00286     break;
00287   case DECADE:
00288     if (_step[Nest] == 0.) {
00289       _step[Nest] = 1.;
00290     }else{
00291     }
00292     _step[Nest] = pow(10., 1./_step[Nest]);
00293     _linswp[Nest] = false;
00294     break;
00295   };
00296   
00297   if (_step[Nest] == 0.) {      // prohibit log sweep from 0
00298     _step[Nest] = _stop[Nest] - _start[Nest];
