00121 void EVAL_BM_POLY::tr_eval(ELEMENT* d)const
00122 {
00123   double x = ioffset(d->_y[0].x);
00124   double f0 = 0.;
00125   double f1 = 0.;
00126   for (size_t i=_c.size()-1; i>0; --i) {
00127     f0 += _c[i];
00128     f0 *= x;
00129     f1 *= x;
00130     f1 += _c[i]*int(i);
00131   }
00132   f0 += _c[0];
00133 
00134   if (_abs && f0 < 0) {
00135     f0 = -f0;
00136     f1 = -f1;
00137   }
00138 
00139   if (f0 > _max) {
00140     f0 = _max;
00141     f1 = 0;
00142   }else if (f0 < _min) {
00143     f0 = _min;
00144     f1 = 0;
00145   }
00146 
00147   d->_y[0] = FPOLY1(x, f0, f1);
00148   tr_final_adjust(&(d->_y[0]), d->f_is_value());
00149 }

00152 {
00153   unsigned start = cmd.cursor();
00154   unsigned here = cmd.cursor();
00155   for (;;) {
00156     unsigned old_here = here;
00157     PARAMETER<double> val;
00158     cmd >> val;
00159     if (cmd.stuck(&here)) {
00160       // no more, graceful finish
00161       break;
00162     }else{
00163       if (cmd.match1('=')) {
00164         // got one that doesn't belong, back up
00165         cmd.reset(old_here);
00166         break;
00167       }else{
00168         _c.push_back(val);
00169       }
00170     }
00171   }
00172   if (cmd.gotit(start)) {
00173   }else{
00174     untested();
00175   }
00176   return cmd.gotit(start);
