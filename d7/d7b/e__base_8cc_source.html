00071 double CKT_BASE::ac_probe_num(const std::string& what)const
00072 {
00073   size_t length = what.length();
00074   mod_t modifier = mtNONE;
00075   bool want_db = false;
00076   char parameter[BUFLEN+1];
00077   strcpy(parameter, what.c_str());
00078 
00079   if (length > 2  &&  Umatch(&parameter[length-2], "db ")) {
00080     want_db = true;
00081     length -= 2;
00082   }
00083   if (length > 1) { // selects modifier based on last letter of parameter
00084     switch (fix_case(parameter[length-1])) {
00085       case 'm': modifier = mtMAG;   length--;   break;
00086       case 'p': modifier = mtPHASE; length--;   break;
00087       case 'r': modifier = mtREAL;  length--;   break;
00088       case 'i': modifier = mtIMAG;  length--;   break;
00089       default:  modifier = mtNONE;              break;
00090     }
00091   }
00092   parameter[length] = '\0'; // chop
00093   
00094   // "p" is "what" with the modifier chopped off.
00095   // Try that first.
00096   XPROBE xp = ac_probe_ext(parameter);
00097 
00098   // If we don't find it, try again with the full string.
00099   if (!xp.exists()) {
00100     xp = ac_probe_ext(what);
00101     if (!xp.exists()) {
00102       // Still didn't find anything.  Print "??".
00103     }else{
00104       untested();
00105       // The second attempt worked.
00106     }
00107   }
00108   return xp(modifier, want_db);
00109 }
