00037   bool delete_one_name(const std::string& name, CARD_LIST* Scope)const
00038   {
00039     assert(Scope);
00040     
00041     std::string::size_type dotplace = name.find_first_of(".");
00042     if (dotplace != std::string::npos) {
00043       // has a dot, look deeper
00044       // Split the name into two parts:
00045       // "container" -- where to look (all following the dot)
00046       // "dev_name" -- what to look for (all before the dot)
00047       std::string dev_name  = name.substr(dotplace+1, std::string::npos);
00048       std::string container = name.substr(0, dotplace);
00049       // container name must be exact match
00050       CARD_LIST::iterator i = Scope->find_(container);
00051       if (i == Scope->end()) {
00052         // can't find "container" (probably .subckt) - no match
00053         // try reverse
00054         dotplace = name.find_last_of(".");
00055         container = name.substr(dotplace+1, std::string::npos);
00056         dev_name  = name.substr(0, dotplace);
00057         // container name must be exact match
00058         i = Scope->find_(container);
00059       }else{
00060       }
00061       if (i == Scope->end()) {
00062         // can't find "container" (probably .subckt) - no match
00063         return false;
00064       }else if (!dynamic_cast<MODEL_SUBCKT*>(*i)) {
00065         // found a match, but it isn't a container (subckt)
00066         return false;
00067       }else{
00068         // found the container, look inside
00069         return delete_one_name(dev_name, (**i).subckt());
00070       }
00071       unreachable();
00072     }else{
00073       // no dots, look here
00074       if (name.find_first_of("*?") != std::string::npos) {
00075         // there's a wild card.  do linear search for all matches
00076         bool didit = false;
00077         {
00078           CARD_LIST::iterator i = Scope->begin();
00079           while (i != Scope->end()) {
00080             CARD_LIST::iterator old_i = i++;
00081             // ^^^^^^^^^^^^ move i past the item being deleted
00082             if (wmatch((**old_i).short_label(), name)) {
00083               Scope->erase(old_i);
00084               didit = true;
00085             }
00086           }
00087         }
00088         return didit;
00089       }else{
00090         // no wild card.  fast search for one exact match
00091         CARD_LIST::iterator i = Scope->find_(name);
00092         if (i != Scope->end()) {
00093           Scope->erase(i);
00094           return true;
00095         }else{
00096           return false;
00097         }
00098       }
00099       unreachable();
00100     }
00101     unreachable();
00102     return false;
00103   }
