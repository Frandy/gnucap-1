00030 CS& CS::umatch(const std::string& s)
00031 {
00032   unsigned start = cursor();
00033   skipbl();
00034   unsigned begin_match = cursor();
00035   const char* str2 = s.c_str();
00036   bool optional = 0;
00037 
00038   for (;;) {
00039     if ((!*str2) || (*str2 == '|')) {
00040       _ok = true;
00041       break;
00042     }else if ((str2[0] == '\\') && (peek() == str2[1])) {
00043       skip();
00044       str2 += 2;
00045     }else if ((!optional) && (*str2 == '{')) {
00046       ++str2;
00047       optional = true;
00048     }else if ((optional) && (*str2 == '}')) {
00049       ++str2;
00050       optional = false;
00051     }else if ((*str2 == ' ') && is_term()) {
00052       // blank in ref string matches anything that delimits tokens
00053       skipbl();
00054       ++str2;
00055     }else if (peek() == *str2) {
00056       skip();
00057       ++str2;
00058     }else if ((OPT::case_insensitive) && (tolower(peek()) == tolower(*str2))) {
00059       skip();
00060       ++str2;
00061     }else if (optional) {
00062       while (*str2 != '}') {
00063         ++str2;
00064       }
00065     }else{
00066       // mismatch
00067       const char* bar = strchr(str2, '|');
00068       if (bar && (bar[-1] != '\\')) {
00069         str2 = bar+1;
00070         reset(start);
00071       }else{
00072         _ok = false;
00073         break;
00074       }
00075     }
00076   }
00077 
00078   if (_ok) {
00079     _begin_match = begin_match;
00080     _end_match = cursor();
00081     skipcom();
00082     _ok = true;
00083   }else{
00084     reset(start);
00085     _ok = false;
00086   }
00087   return *this;
00088 }
