00621 void EVAL_BUILT_IN_DIODE_Cj::tr_eval(ELEMENT* d)const
00622 {
00623   assert(d);
00624   DEV_BUILT_IN_DIODE* p = prechecked_cast<DEV_BUILT_IN_DIODE*>(d->owner());
00625   assert(p);
00626   const COMMON_BUILT_IN_DIODE* c = prechecked_cast<const COMMON_BUILT_IN_DIODE*>(p->common());
00627   assert(c);
00628   const SDP_BUILT_IN_DIODE* s = prechecked_cast<const SDP_BUILT_IN_DIODE*>(c->sdp());
00629   assert(s);
00630   const MODEL_BUILT_IN_DIODE* m = prechecked_cast<const MODEL_BUILT_IN_DIODE*>(c->model());
00631   assert(m);
00632 
00633     double& volts = d->_y[0].x;
00634     trace1(d->long_label().c_str(), volts);
00635     
00636     double cb;
00637     if (c->cj_adjusted != 0.) {
00638       if (volts < m->fc * m->pb) {
00639         cb = c->cj_adjusted / pow(1. - (volts / m->pb),  m->mj);
00640       }else{
00641         cb = (c->cj_adjusted / pow(1. - m->fc, 1. + m->mj))
00642           * (1. - m->fc*(1.+m->mj) + (volts/m->pb)*m->mj);
00643       }
00644     }else{
00645       cb = 0.;
00646     }
00647     assert(cb >= 0.);
00648     
00649     double csw;
00650     if (c->cjsw_adjusted != 0.) {
00651       if (volts < m->fc * m->pbsw) {
00652         csw = c->cjsw_adjusted / pow(1. - (volts / m->pbsw),  m->mjsw);
00653       }else{
00654         csw = (c->cjsw_adjusted / pow(1. - m->fc, 1. + m->mjsw))
00655           * (1. - m->fc*(1.+m->mjsw) + (volts/m->pbsw)*m->mjsw);
00656       }
00657     }else{
00658       csw = 0.;
00659     }
00660     assert(csw >= 0.);
00661     
00662     double ctt;
00663     if (m->tt != 0.) {
00664       ctt = p->_gd * m->tt;
00665     }else{
00666       ctt = 0.;
00667     }
00668     assert(ctt >= 0.);
00669     
00670     trace4("", cb, csw, ctt, cb+csw+ctt);
00671     d->_y[0].f1 = cb + csw + ctt;
00672     if (d->_sim->analysis_is_tran_dynamic()) {
00673       const STORAGE* dd = prechecked_cast<const STORAGE*>(d);
00674       assert(dd);
00675       double cap = (d->_y[0].f1 + dd->_y[1].f1) / 2;
00676       d->_y[0].f0 = (d->_y[0].x - dd->_y[1].x) * cap + dd->_y[1].f0;
00677     }else{
00678       assert(d->_sim->analysis_is_static() || d->_sim->analysis_is_restore());
00679       d->_y[0].f0 = d->_y[0].x * d->_y[0].f1;
00680     }
00681     trace3(d->long_label().c_str(), d->_y[0].x, d->_y[0].f0, d->_y[0].f1);
00682 }

00684 static EVAL_BUILT_IN_DIODE_Yj Eval_Yj(CC_STATIC);
00685 void EVAL_BUILT_IN_DIODE_Yj::tr_eval(ELEMENT* d)const
00686 {
00687   assert(d);
00688   DEV_BUILT_IN_DIODE* p = prechecked_cast<DEV_BUILT_IN_DIODE*>(d->owner());
00689   assert(p);
00690   const COMMON_BUILT_IN_DIODE* c = prechecked_cast<const COMMON_BUILT_IN_DIODE*>(p->common());
00691   assert(c);
00692   const SDP_BUILT_IN_DIODE* s = prechecked_cast<const SDP_BUILT_IN_DIODE*>(c->sdp());
00693   assert(s);
00694   const MODEL_BUILT_IN_DIODE* m = prechecked_cast<const MODEL_BUILT_IN_DIODE*>(c->model());
00695   assert(m);
00696 
00697     FPOLY1& y = d->_y[0];
00698     double volts = y.x;
00699     double amps  = y.f0;
00700     trace2(d->long_label().c_str(), volts, amps);
00701     
00702     int flags = (m->flags & USE_OPT) ? OPT::diodeflags : m->flags;
00703     double tempratio = (d->_sim->_temp_c+P_CELSIUS0) / (m->_tnom_c+P_CELSIUS0);
00704     double vt = P_K_Q * (d->_sim->_temp_c+P_CELSIUS0) * m->n_factor;
00705     region_t oldregion = p->_region;
00706     p->_isat = c->is_adjusted * pow(tempratio, m->xti)
00707       * exp((m->eg/vt) *(tempratio-1));
00708     trace4("", tempratio, vt, oldregion, p->_isat);
00709     
00710     if (m->mos_level > 0 || flags & 0040) { // Spice style limiting
00711       double vcrit = vt * log(vt / (M_SQRT2 * p->_isat));
00712       double vold = d->_y1.f0;
00713       if((volts > vcrit) && (std::abs(volts - vold) > (vt + vt))) {
00714         if(vold > 0) {
00715           double arg = 1 + (volts - vold) / vt;
00716           if(arg > 0) {
00717             volts = vold + vt * log(arg);
00718           }else{
00719             volts = vcrit;
00720           }
00721         }else{
00722           volts = vt *log(volts/vt);
00723         }
00724       }else{
00725         // leave volts as is
00726       }
00727     }
00728     
00729     if (m->mos_level > 0) {
00730       switch (m->mos_level) {
00731       case 1:
00732       case 2:
00733       case 3:
00734       case 6:
00735       case 4:
00736       case 5:
00737         if (volts <= 0.) {
00738           p->_region = REVERSE;
00739           y.f1 = p->_isat / vt + OPT::gmin;
00740           y.f0 = y.f1 * volts;
00741         }else{
00742           p->_region = FORWARD;
00743           double ev = exp(volts/vt);
00744           y.f1 = p->_isat * ev / vt + OPT::gmin;
00745           y.f0 = p->_isat * (ev - 1) + OPT::gmin * volts;
00746         }
00747         break;
00748       case 7:
00749       case 8:
00750         if (volts < .5) {
00751           p->_region = REVERSE;
00752           double ev = exp(volts/vt);
00753           y.f1 = p->_isat * ev / vt + OPT::gmin;
00754           y.f0 = p->_isat * (ev - 1) + OPT::gmin * volts;
00755         }else{
00756           p->_region = FORWARD;
00757           double ev = exp(.5/vt);
00758           double t0 = p->_isat * ev / vt;
00759           y.f1 = t0 + OPT::gmin;
00760           y.f0 = p->_isat * (ev - 1) + t0 * (volts - .5) + OPT::gmin * volts;
00761         }
00762         break;
00763       default:
00764         unreachable();
00765         y.f1 = OPT::gmin;
00766         y.f0 = volts * y.f1;
00767       }
00768     }else if (flags & 0040) { // exact Spice model
00769       if (volts >= -3*vt) { // forward and weak reversed
00770         double evd = exp(volts/vt);
00771         y.f0 = p->_isat * (evd-1);
00772         y.f1 = p->_isat * evd/vt;
00773       }else if (has_good_value(m->bv) || volts >= m->bv) {
00774         double arg = 3 * vt / (volts * M_E); // strong reversed
00775         arg = arg * arg * arg;
00776         y.f0 = -p->_isat * (1+arg);
00777         y.f1 = p->_isat * 3 * arg / volts;
00778       }else{
00779         incomplete();
00780         double evrev = exp(-(m->bv+volts)/vt);
00781         y.f0 = -p->_isat * evrev;
00782         y.f1 = p->_isat * evrev / vt;
00783       }
00784       y.f0 += OPT::gmin * volts;
00785       y.f1 += OPT::gmin;
00786     }else{
00787        if (c->off  &&  d->_sim->is_initial_step()) { /*initially guess off*/
00788         p->_region = INITOFF;
00789         y.f1 = 0.;
00790         y.f0 = 0.;
00791         if (flags & 0020) {
00792           untested();
00793           y.f1 = OPT::gmin;
00794         }
00795         trace2("initoff", y.f0, y.f1);
00796       }else if (volts <= 0. /* &&  amps < 0.*/) {         /* reverse biased */
00797         p->_region = REVERSE;                     /* x = volts, f(x) = amps */
00798         if (flags & 0010) {
00799           untested();
00800           y.f1 = y.f0 = 0.;
00801         }else{
00802           double expterm = p->_isat * exp(volts/vt);    
00803           y.f0 = expterm - p->_isat;/* i = f(x) = _isat * (exp(volts/vt)-1) */
00804           y.f1 = expterm / vt;      /* f'(x) = (_isat/vt) * exp(volts/vt)   */
00805         }
00806         
00807         if (flags & 0002) {     // g = gmin, maintain actual current
00808           y.f1 += OPT::gmin;    // 3 is a resistor, R=1/gmin
00809           y.f0 += OPT::gmin * volts;
00810         }
00811         if (flags & 0004) {     // 5 is a resistor, R=vt/_isat
00812           double x = p->_isat / vt;
00813           y.f1 += x;
00814           y.f0 += x * volts;
00815         }
00816         if (flags & 0001) {
00817           //y.f0 = y.f1 * volts;        // a resistor, R=1/f1
00818         }
00819         
00820         trace2("reverse", y.f0, y.f1);
00821       }else if (volts >= 0.  &&  amps >= 0.) {            /* forward biased */
00822                                     /* x = amps, f(x) = volts */
00823         /* derivation: */           /* if f(x) = log(u): f'(x)=(1/u)(du/dx) */
00824         /* poly1 r; */
00825         /* r.f0 = vt * log(amps/p->_isat +1.); */
00826         /* r.f1 = vt / (_isat + amps); */
00827         /* y.f1 = 1. / r.f1; */
00828         /* y.f0 = amps - r.f0*y.f1 + volts*y.f1; */
00829         
00830         p->_region = FORWARD;
00831         y.f1 = (p->_isat + amps) / vt;
00832         y.f0 = amps - log(amps/p->_isat +1.)*(p->_isat + amps) + volts*y.f1;
00833         trace2("forward", y.f0, y.f1);
00834       }else{                        /* non-converged, inconsistent          */
00835         p->_region = UNKNOWN;       /* volts and amps have different signs  */
00836         y.f1 = p->_isat/vt;         /* guess that the voltage should be 0   */
00837         y.f0 = 0.;                  /* (it usually is very close)           */
00838         if (flags & 0001) {         /* use the correct value there          */
00839           y.f0 = volts * y.f1;
00840         }
00841         trace2("unknown", y.f0, y.f1);
00842       }
00843       y.f1 += c->gparallel_adjusted;
00844       y.f0 += c->gparallel_adjusted * volts;
00845       
00846       if (oldregion != p->_region  &&  OPT::dampstrategy & dsDEVLIMIT) {
00847         d->_sim->_fulldamp = true;
00848         error(bTRACE, p->long_label() + ":device limit damp\n");
00849       }
00850       if (flags & 0100) {               // twist g to guarantee g >= gmin
00851         if (y.f1 < OPT::gmin) { // without changing i
00852           y.f1 = OPT::gmin;
00853           untested();
00854         }else{
00855           untested();
00856         }
00857       }
00858       if (flags & 0200) {               // add a gmin in parallel
00859         y.f1 += OPT::gmin;
00860         y.f0 += OPT::gmin * volts;
00861         untested();
00862       }
00863       if (flags & 0400) {               // linearize .. shift I to pass thru 0
00864         untested();
00865         y.f0 = y.f1 * volts;
00866       }
00867     }
00868     trace3(d->long_label().c_str(), y.x, y.f0, y.f1);
00869     p->_gd = y.f1;
00870 }
