00136 void EVAL_BM_POSY::tr_eval(ELEMENT* d)const
00137 {
00138   double x_raw = ioffset(d->_y[0].x);
00139   trace1("before", x_raw);
00140   double x = (x_raw < 0)
00141     ? ((_odd || _even) ? -x_raw : 0.)
00142     : x_raw;
00143   assert(x >= 0);
00144   trace1("rev", x);
00145 
00146   double f0 = 0.;
00147   double f1 = 0.;
00148   if (x > 0) {
00149     for (std::vector<std::pair<PARAMETER<double>,PARAMETER<double> > >::
00150            const_iterator p = _table.begin();  p != _table.end();  ++p) {
00151       double coeff = p->second * pow(x, p->first-1);
00152       f1 += coeff * p->first;
00153       f0 += coeff * x;
00154     }
00155   }else{
00156     assert(x == 0);
00157     for (std::vector<std::pair<PARAMETER<double>,PARAMETER<double> > >::
00158            const_iterator p = _table.begin();  p != _table.end();  ++p) {
00159       if (p->first == 0.) {
00160         f0 += 1;
00161         untested();
00162       }
00163     }
00164   }
00165 
00166   trace3("", x, f0, f1);
00167 
00168   if (_odd && x_raw < 0) {
00169     f0 = -f0;
00170   }
00171   if (_even && x_raw < 0) {
00172     f1 = -f1;
00173     untested();
00174   }
00175   trace3("after", x, f0, f1);
00176 
00177   if (_abs && f0 < 0) {
00178     f0 = -f0;
00179     f1 = -f1;
00180     untested();
00181   }
00182 
00183   if (f0 > _max) {
00184     f0 = _max;
00185     f1 = 0;
00186     untested();
00187   }else if (f0 < _min) {
00188     f0 = _min;
00189     f1 = 0;
00190     untested();
00191   }
00192 
00193   d->_y[0] = FPOLY1(x_raw, f0, f1);
00194   tr_final_adjust(&(d->_y[0]), d->f_is_value());
00195   trace3("fa", d->_y[0].x, d->_y[0].f0, d->_y[0].f1);
00196 }

00200   unsigned start = cmd.cursor();

00202   for (;;) {
00203     unsigned start_of_pair = here;
00204     std::pair<PARAMETER<double>, PARAMETER<double> > p;
00205     cmd >> p.second; // value
00206     if (cmd.stuck(&here)) {
00207       // no more, graceful finish
00208       break;
00209     }else{
00210       cmd >> p.first; // key
00211       if (cmd.stuck(&here)) {
00212         // ran out, but already have half of the pair
00213         // back up one, hoping somebody else knows what to do with it
00214         cmd.reset(start_of_pair);
00215         break;
00216       }else{
00217         _table.push_back(p);
00218       }
00219     }
00220   }
00221   if (cmd.gotit(start)) {
00222   }else{
