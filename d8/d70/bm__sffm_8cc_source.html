00155 TIME_PAIR EVAL_BM_SFFM::tr_review(COMPONENT* d)
00156 {
00157   double time = d->_sim->_time0 + d->_sim->_dtmin * .01;
00158   double old_time;
00159   double N = 0;
00160   double old_N;
00161   trace2("", time, N);
00162   do {
00163     // simple fixed point iteration to find peaks and zero crossings
00164     old_time = time;
00165     old_N = N;
00166     double mod = (_modindex * sin(M_TWO_PI * _signal * time));
00167     double inst_freq = _carrier * (1 + mod / (M_TWO_PI * _carrier * time));
00168     if (N == 0) {
00169       if (_peak && _zero) {
00170         N = floor(time * 4 * inst_freq + 1);
00171       }else if (_peak) {untested();
00172         N = floor(time * 2 * inst_freq + .5) + .5;
00173       }else if (_zero) {untested();
00174         N = floor(time * 2 * inst_freq + 1);
00175       }else{
00176         N = 0;
00177       }
00178     }else{
00179     }
00180     if (_peak && _zero) {
00181       time = N / (4 * inst_freq);
00182     }else if (_peak) {untested();
00183       time = N / (2 * inst_freq);
00184     }else if (_zero) {untested();
00185       time = N / (2 * inst_freq);
00186     }else{
00187       time = NEVER;
00188     }
00189     trace2("", time, N);
00190   } while (std::abs(dn_diff(time, old_time)) > 0);
00191 
00192   d->_time_by.min_error_estimate(d->_sim->_time0 + 1. / (_samples * _carrier));
00193   d->_time_by.min_event(old_time);
00194   
00195   return d->_time_by;
00196 }
