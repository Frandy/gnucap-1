00202   if (d0 == NOT_INPUT) {

00205   }else{
00206     u[0] = .5;  // h[0] / (2*h[0])
00207     z[0] = b[0] / (2*h[0]);

00209   for (int i=1; i<_n; ++i) {
00210     double p = 2*(h[i]+h[i-1]) - h[i-1]*u[i-1]; // pivot
