00148 void EVAL_BM_PULSE::tr_eval(ELEMENT* d)const
00149 {
00150   double time = d->_sim->_time0;
00151   if (0 < _period && _period < BIGBIG) {
00152     //time = fmod(time,_period);
00153     if (time > _delay) {
00154       time = fmod(time - _delay, _period) + _delay;
00155     }else{
00156     }
00157   }else{
00158   }
00159   double ev = 0; // effective value
00160   if (time >= _delay+_rise+_width+_fall) {      /* past pulse   */
00161     ev = _iv;
00162   }else if (time >= _delay+_rise+_width) {      /* falling      */
00163     double interp = (time - (_delay+_rise+_width)) / _fall;
00164     ev = _pv + interp * (_iv - _pv);
00165   }else if (time >= _delay + _rise) {           /* pulse val    */
00166     ev = _pv;
00167   }else if (time >= _delay) {                   /* rising       */
00168     double interp = (time - _delay) / _rise;
00169     ev = _iv + interp * (_pv - _iv);
00170   }else{                                        /* init val     */
00171     ev = _iv;
00172   }
00173   //d->q_accept();
00174   tr_finish_tdv(d, ev);
00175 }

00181   double raw_time = time;

00194   }else if (time >= _delay+_rise+_width) {              /* falling      */
00195     d->_time_by.min_event(_delay + _rise + _width + _fall + time_offset);
00196   }else if (time >= _delay + _rise) {                   /* pulse val    */
00197     d->_time_by.min_event(_delay + _rise + _width + time_offset);
00198   }else if (time >= _delay) {                           /* rising       */
00199     d->_time_by.min_event(_delay + _rise + time_offset);
00200   }else{                                                /* init val     */
