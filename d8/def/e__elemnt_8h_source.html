00169 inline double ELEMENT::dampdiff(double* v0, const double& v1)
00170 {
00171   //double diff = v0 - v1;
00172   assert(v0);
00173   assert(*v0 == *v0);
00174   assert(v1 == v1);
00175   double diff = dn_diff(*v0, v1);
00176   assert(diff == diff);
00177   if (!_sim->is_advance_or_first_iteration()) {
00178     diff *= _sim->_damp;
00179     *v0 = v1 + diff;
00180   }else{
00181   }
00182   return mfactor() * ((_sim->is_inc_mode()) ? diff : *v0);
00183 }
