00042 bool SIM::solve(OPT::ITL itl, TRACE trace)
00043 {
00044   converged = false;
00045   int convergedcount = 0;
00046   
00047   _sim->reset_iteration_counter(iSTEP);
00048   advance_time();
00049 
00050   _sim->_damp = OPT::dampmax;
00051  
00052   do{
00053     if (trace >= tITERATION) {
00054       print_results(static_cast<double>(-_sim->iteration_number()));
00055     }
00056     set_flags();
00057     clear_arrays();
00058     finish_building_evalq();
00059     
00060     _sim->count_iterations(iPRINTSTEP);
00061     _sim->count_iterations(iSTEP);
00062     _sim->count_iterations(_sim->_mode);
00063     _sim->count_iterations(iTOTAL);
00064     
00065     evaluate_models();
00066 
00067     if (converged) {
00068       if (_sim->_limiting) {
00069         error(bDEBUG, "converged beyond limit, resetting limit\n");
00070         _sim->set_limit();
00071         convergedcount = 0;
00072       }else{
00073         ++convergedcount;
00074       }
00075     }else{
00076       convergedcount = 0;
00077     }
00078     if (convergedcount <= OPT::itermin) {
00079       converged = false;
00080     }
00081       
00082     if (!converged || !OPT::fbbypass || _sim->_damp < .99) {
00083       set_damp();
00084       load_matrix();
00085       solve_equations();
00086     }
00087   }while (!converged && !_sim->exceeds_iteration_limit(itl));
00088 
00089   return converged;
00090 }

00143 void SIM::advance_time(void)
00144 {
00145   ::status.advance.start();
00146   static double last_iter_time;
00147   if (_sim->_time0 > 0) {
00148     if (_sim->_time0 > last_iter_time) {        /* moving forward */
00149       notstd::copy_n(_sim->_v0, _sim->_total_nodes+1, _sim->_vt1);
00150       CARD_LIST::card_list.tr_advance();
00151     }else{                              /* moving backward */
00152       /* don't save voltages.  They're wrong! */
00153       /* instead, restore a clean start for iteration */
00154       notstd::copy_n(_sim->_vt1, _sim->_total_nodes+1, _sim->_v0);
00155       CARD_LIST::card_list.tr_regress();
00156     }
00157   }else{
00158     CARD_LIST::card_list.dc_advance();
00159   }
00160   last_iter_time = _sim->_time0;
00161   ::status.advance.stop();
00162 }
00163 /* last_iter_time is initially 0 by C definition.
00164  * On subsequent runs it will start with an arbitrary positive value.
00165  * _sim->_time0 starts at either 0 or the ending time of the last run.
00166  * In either case, (time0 > last_iter_time) is false on the first step.
00167  * This correctly results in "don't save voltages..."
00168  */
