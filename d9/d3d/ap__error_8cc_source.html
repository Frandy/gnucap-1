00031 /* syntax_check: handle syntax errors
00032  * called on parsing an input string when nothing else matches.
00033  * if the rest of the line is nothing, just return
00034  * if comment, increment *cnt, so what is left is a valid comment string
00035  * otherwise, it is an error (the arrow pointing at the offending word)
00036  */
00037 CS & CS::check(int badness, const std::string& message)
00038 {
00039   skipbl();
00040   switch (peek()) {
00041   case '/': _ok = umatch("//"); skip(); break;
00042   case ';':
00043   case '\'':    itested(); _ok = true;  skip();     break;
00044   case '\0':                _ok = true;              break;
00045   default:      _ok = false; warn(badness, message); break;
00046   }
00047   return *this;
00048 }
00049 /*--------------------------------------------------------------------------*/
00050 /* syntax_msg: print syntax error message
00051  * echo a portion of the input with an arrow pointing to the current place
00052  */
00053 CS & CS::warn(int badness, unsigned spot, const std::string& message)
00054 {
00055   if (badness >= OPT::picky) {
00056     if (spot < 40) {
00057       IO::error << _cmd.substr(0,70) << '\n';
00058       IO::error.tab(spot);
00059     }else{
00060       IO::error << _cmd.substr(0,15) << " ... " << _cmd.substr(spot-20, 56) << '\n';
00061       IO::error.tab(40);
00062     }
00063     IO::error << "^ ? " + message + '\n';
00064   }else{
00065   }
00066   return *this;
00067 }
