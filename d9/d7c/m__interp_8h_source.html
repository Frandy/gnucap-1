00029 /* interpolate:  linear interpolation on a table.
00030  * Keys must be sorted in increasing order.
00031  */
00032 template <class Iterator>
00033 FPOLY1 interpolate(Iterator begin, Iterator end, double x,
00034                    double below, double above)
00035 {
00036   double f1 = NOT_VALID;
00037   double f0 = NOT_VALID;
00038   if (begin == end) {
00039     untested();
00040     throw Exception("interpolate table is empty");
00041   }
00042   --end;
00043   if (begin == end) { // only 1 entry -- constant
00044     untested();
00045     f1 = (x < (*begin).first)
00046       ? ((below != NOT_INPUT) ? below : 0.)
00047       : ((above != NOT_INPUT) ? above : 0.);
00048     f0 = (*begin).second + (x - (*begin).first) * f1;
00049   }else{
00050     ++begin;
00051 
00052     // x is the search key
00053     // xx is the search key converted to a "pair" as required by upper_bound
00054     // upper might point to a match for the key, exact match
00055     // if not, it points just above it, no exact match
00056     // lower points just below where a match would go
00057     // so the real match is between upper and lower
00058     DPAIR    xx(x,BIGBIG);
00059     Iterator upper = upper_bound(begin, end, xx);
00060     Iterator lower = upper-1;
00061 
00062     // set f1 (derivative)
00063     if ((upper == end) && (x > (*upper).first) && (above != NOT_INPUT)) {
00064       // x is out of bounds, above
00065       lower = upper;
00066       if (above != 0.) {
00067         untested();
00068       }
00069       f1 = above;
00070     }else if ((upper == begin) && (x < (*lower).first) && (below!=NOT_INPUT)) {
00071       // x is out of bounds, below
00072       untested();
00073       if (below != 0.) {
00074         untested();
00075       }else{
00076         untested();
00077       }
00078       f1 = below;
00079     }else if ((*upper).first <= (*lower).first) {untested();
00080       throw Exception("interpolate table is not sorted or has duplicate keys");
00081       f1 = 0.;
00082     }else{
00083       // ordinary interpolation
00084       assert((*upper).first != (*lower).first);
00085       f1 = ((*upper).second-(*lower).second) / ((*upper).first-(*lower).first);
00086     }
00087

00089   }

00092   return FPOLY1(x, f0, f1);
