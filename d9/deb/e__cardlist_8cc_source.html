00457 void CARD_LIST::map_subckt_nodes(const CARD* model, const CARD* owner)
00458 {
00459   assert(model);
00460   assert(model->subckt());
00461   assert(model->subckt()->nodes());
00462   assert(owner);
00463   //assert(owner->subckt());
00464   //assert(owner->subckt() == this);
00465   trace0(model->long_label().c_str());
00466   trace0(owner->long_label().c_str());
00467 
00468   int num_nodes_in_subckt = model->subckt()->nodes()->how_many();
00469   int* map = new int[num_nodes_in_subckt+1];
00470   {
00471     map[0] = 0;
00472     // self test: verify that port node numbering is correct
00473     for (int port = 0; port < model->net_nodes(); ++port) {
00474       assert(model->n_(port).e_() <= num_nodes_in_subckt);
00475       //assert(model->n_(port).e_() == port+1);
00476       trace3("ports", port, model->n_(port).e_(), owner->n_(port).t_());
00477     }
00478     {
00479       // take care of the "port" nodes (external connections)
00480       // map them to what the calling circuit wants
00481       int i=0;
00482       for (i=1; i <= model->net_nodes(); ++i) {
00483         map[i] = owner->n_(i-1).t_();
00484         trace3("ports", i, map[i], owner->n_(i-1).t_());
00485       }
00486     
00487       // get new node numbers, and assign them to the remaining
00488       for (assert(i==model->net_nodes() + 1); i <= num_nodes_in_subckt; ++i) {
00489         // for each remaining node in card_list
00490         map[i] = CKT_BASE::_sim->newnode_subckt();
00491         trace2("internal", i, map[i]);
00492       }
00493     }
00494   }
00495   // "map" now contains a translation list,
00496   // from subckt local numbers to matrix index numbers
00497 
00498   // The node list (_nm) in an instance of a subckt does not exist.
00499   // Device nodes (type node_t) points to the NODE in the parent.
00500   // Mapping is done in node_t.
00501 
00502   // scan the list, map the nodes
00503   for (CARD_LIST::iterator ci = begin(); ci != end(); ++ci) {
00504     // for each card in card_list
00505     if ((**ci).is_device()) {
00506       for (int ii = 0;  ii < (**ci).net_nodes();  ++ii) {
00507         // for each connection node in card
00508         (**ci).n_(ii).map_subckt_node(map, owner);
00509       }
00510     }else{
00511       assert(dynamic_cast<MODEL_CARD*>(*ci));
00512     }
00513   }
00514 }
