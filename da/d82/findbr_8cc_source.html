00032 /* findbranch: find a matching label, by (ugh) linear search
00033  *      label to look for is in command line (cmd).
00034  *      start "here".  Look only after "here".
00035  * return pointer to next match if exists (and eat input)
00036  *        pointer to a non-existent branch if no match (don't eat input)
00037  * caution: caller must check return value before using
00038  */
00039 CARD_LIST::fat_iterator findbranch(CS& cmd, CARD_LIST::fat_iterator here)
00040 {
00041   unsigned save = cmd.cursor();
00042 
00043   char labelwanted[BUFLEN+1];
00044   cmd.ctostr(labelwanted, BUFLEN, TOKENTERM);
00045   
00046   if (!labelwanted[0]) {                // nothing to look for
00047     cmd.reset(save);
00048     return here.end();
00049   }else{
00050   }
00051   
00052   char* local_part;         // part before last dot, the thing inside
00053   char* last_part;          // part after last dot, top level
00054   {
00055     char* dot = strrchr(labelwanted,'.');
00056     if (dot) {itested();                    // split the string into 2 parts at the last dot
00057       *dot = '\0';          // before is the new "local_part", shortened
00058       last_part = dot + 1;  // after is the "last_part".
00059       local_part = labelwanted;
00060     }else{
00061       last_part = labelwanted;
00062       local_part = NULL;
00063     }
00064   }
00065 
00066   for (;;) {
00067     if (here.is_end()) {
00068       // at the end of the list - done, fails.
00069       cmd.reset(save);
00070       return here;
00071     }else if (wmatch((**here).short_label(), last_part)) { 
00072       // last_part matches
00073       if (!local_part) {
00074         // requested a simple name, found it .. done.
00075         return here;
00076       }else{untested();
00077         // there are dots, so look inside subckt
00078         untested();
00079         if ((**here).subckt()) {untested();
00080           untested();
00081           // has a subckt, and its name matches, doing fine
00082           CS want(CS::_STRING, local_part); // recursion
00083           CARD_LIST::fat_iterator subbrh=findbranch(want,(**here).subckt());
00084           if (!subbrh.is_end()) {untested();
00085             // found it in a subckt
00086             return subbrh;
00087           }else{untested();
00088             // didn't find anything in this subckt
00089             // keep looking for another with the same name
00090             // why?
00091             ++here;
00092           }
00093         }else{untested(); 
00094           // no subckt
00095           // keep looking for something with this name that has a subckt
00096           // why?
00097           untested();
00098           ++here;
00099         }
00100       }
00101     }else{
00102       // label doesn't match
00103       // keep looking for one that does.  (linear search)
00104       ++here;
00105     }
00106   }
00107 }
