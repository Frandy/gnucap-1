00354     int lastchangeiter=_n[OUTNODE]->d_iter();/* iteration # when it changed */

00356     trace2(_n[OUTNODE]->failure_mode().c_str(), OUTNODE, _n[OUTNODE]->quality());
00357

00359       _n[ii]->to_logic(m);
00360       if (_n[ii]->quality() < _quality) {

00362         _failuremode = _n[ii]->failure_mode();
00363       }else{
00364       }
00365       if (_n[ii]->d_iter() >= lastchangeiter) {
00366         lastchangeiter = _n[ii]->d_iter();

00368       }else{

00371     }
