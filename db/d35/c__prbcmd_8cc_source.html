00034 void do_probe(CS& cmd, PROBELIST *probes)
00035 {
00036   CKT_BASE::_sim->set_command_none();
00037   enum {aADD, aDELETE, aNEW} action;
00038   SIM_MODE simtype = s_NONE;
00039 
00040   if (cmd.match1('-')) {untested();     /* handle .probe - ac ...... */
00041     action = aDELETE;           /* etc.                      */
00042     cmd.skip();
00043   }else if (cmd.match1('+')) {untested();
00044     action = aADD;
00045     cmd.skip();
00046   }else{                        /* no -/+ means clear, but wait for */
00047     action = aNEW;              /* .probe ac + .....                */
00048   }                             /* which will not clear first       */
00049 
00050   ONE_OF
00051     || Set(cmd, "tr{ansient}", &simtype, s_TRAN)
00052     || Set(cmd, "ac",          &simtype, s_AC)
00053     || Set(cmd, "dc",          &simtype, s_DC)
00054     || Set(cmd, "op",          &simtype, s_OP)
00055     || Set(cmd, "fo{urier}",   &simtype, s_FOURIER)
00056     ;
00057   
00058   if (!simtype) {                       /* must be all simtypes */
00059     if (cmd.is_end()) {                 /* list all */
00060       probes[s_TRAN].listing("tran");
00061       probes[s_AC].listing("ac");
00062       probes[s_DC].listing("dc");
00063       probes[s_OP].listing("op");
00064       probes[s_FOURIER].listing("fourier");
00065     }else if (cmd.umatch("clear ")) {           /* clear all */
00066       for (int ii = sSTART;  ii < sCOUNT;  ++ii) {
00067         probes[ii].clear();
00068       }
00069     }else{itested();                            /* error */
00070       throw Exception_CS("what's this?", cmd);
00071     }
00072   }else{
00073     if (cmd.is_end()) {untested();              /* list */
00074       probes[simtype].listing("");
00075     }else if (cmd.umatch("clear ")) {untested();/* clear */
00076       probes[simtype].clear();
00077     }else{                                      /* add/remove */
00078       CKT_BASE::_sim->init();
00079       if (cmd.match1('-')) {itested();          /* setup cases like: */
00080         action = aDELETE;                       /* .probe ac + ....  */
00081         cmd.skip();
00082       }else if (cmd.match1('+')) {
00083         action = aADD;
00084         cmd.skip();
00085       }else{
00086       }
00087       if (action == aNEW) {                     /* no +/- here or at beg. */
00088         probes[simtype].clear();                /* means clear first      */
00089         action = aADD;
00090       }else{
00091       }
00092       while (cmd.more()) {                      /* do-it */
00093         if (cmd.match1('-')) {                  /* handle cases like:       */
00094           action = aDELETE;                     /* .pr ac +v(7) -e(6) +r(8) */
00095           cmd.skip();
00096         }else if (cmd.match1('+')) {itested();
00097           action = aADD;
00098           cmd.skip();
00099         }else{
00100         }
00101         if (action == aDELETE) {
00102           probes[simtype].remove_list(cmd);
00103         }else{
00104           probes[simtype].add_list(cmd);
00105         }
00106       }
00107     }
00108   }
00109 }
