00137 bool DEV_CAPACITANCE::do_tr()
00138 {
00139   if (using_tr_eval()) {
00140     _y[0].x = tr_input_limited();
00141     tr_eval();
00142   }else{
00143     _y[0].x = tr_input(); // tr_involts();
00144     assert(_y[0].f1 == value());
00145     _y[0].f0 = _y[0].x * _y[0].f1;
00146     assert(converged());
00147   }
00148   store_values();
00149   q_load();
00150 
00151   trace3("q", _y[0].x, _y[0].f0, _y[0].f1);
00152   _i[0] = differentiate(_y, _i, _time, _method_a);
00153   trace3("i", _i[0].x, _i[0].f0, _i[0].f1);
00154   _m0 = CPOLY1(_i[0]);
00155   return converged();

00169 double DEV_CAPACITANCE::tr_probe_num(const std::string& x)const
00170 {
00171   if (Umatch(x, "q{cap} |ch{arge} ")) {
00172     return _y[0].f0;
00173   }else if (Umatch(x, "c{apacitance} ")) {
00174     return _y[0].f1;
00175   }else if (Umatch(x, "dcdt ")) {untested();
00176     return (_y[0].f1 - _y[1].f1) / _dt;
00177   }else if (Umatch(x, "dc ")) {untested();
00178     return (_y[0].f1 - _y[1].f1);
00179   }else if (Umatch(x, "dqdt ")) {
00180     return (_y[0].f0 - _y[1].f0) / _dt;
00181   }else if (Umatch(x, "dq ")) {
00182     return (_y[0].f0 - _y[1].f0);
00183   }else{
00184     return STORAGE::tr_probe_num(x);
00185   }
00186 }

00188 bool DEV_VCCAP::do_tr()
00189 {
00190   _y[0].x = tr_input_limited();
00191   tr_eval();
00192 
00193   store_values();
00194   q_load();
00195 
00196   _y[0].x = tr_outvolts();
00197   _y[0].f1 = _y[0].f0;           // self capacitance
00198   _y[0].f0 = _y[0].x * _y[0].f1; // charge
00199 
00200   _i[0] = differentiate(_y, _i, _time, _method_a);
00201   _m0.x  = _i[0].x;
00202   _m0.c1 = _i[0].f1;
00203   _m0.c0 = _i[0].f0 - _i[0].x * _i[0].f1;
00204   return converged();
00205 }
