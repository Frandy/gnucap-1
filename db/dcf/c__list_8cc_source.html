00033 void list_save(CS& cmd, OMSTREAM out, CARD_LIST* scope)
00034 {
00035   CARD_LIST::card_list.precalc_first();
00036 
00037   //out.setfloatwidth(7);
00038   switch (ENV::run_mode) {
00039   case rPRE_MAIN:
00040     unreachable();
00041     return;
00042   case rPRESET:
00043     /* do nothing */
00044     return;
00045   case rBATCH:          itested();
00046   case rINTERACTIVE:    itested();
00047   case rSCRIPT:
00048     /* keep going */
00049     break;
00050   }
00051 
00052   if (!OPT::language) {
00053     throw Exception("no language");
00054   }else{
00055   }
00056 
00057   (out - IO::mstdout) << head << '\n';
00058 
00059   if (cmd.is_end()) {                   /* no args: list all                */
00060     for (CARD_LIST::const_iterator ci=scope->begin();ci!=scope->end();++ci) {
00061       OPT::language->print_item(out, *ci);
00062     }
00063   }else{                                /* some args: be selective          */
00064     unsigned arg1 = cmd.cursor();
00065     CARD_LIST::fat_iterator ci = (cmd.match1('-')) 
00066       ? CARD_LIST::fat_iterator(scope, scope->begin())
00067       : findbranch(cmd, scope);
00068     if (ci.is_end()) {itested();
00069       throw Exception_CS("can't find", cmd);
00070     }else{
00071     }
00072     
00073     if (cmd.match1('-')) {              /* there is a dash:  a range        */
00074       cmd.skip();
00075       if (cmd.is_end()) {       /* line ends with dash: all to end  */
00076         do {
00077           OPT::language->print_item(out, *ci);
00078         } while (++ci, !ci.is_end());
00079       }else{
00080         CARD_LIST::fat_iterator stop = ci;
00081         stop = findbranch(cmd, ++stop);
00082         if (stop.is_end()) {itested();
00083           throw Exception_CS("can't find", cmd);
00084         }else{
00085           do {
00086             OPT::language->print_item(out, *ci);
00087           } while (ci++ != stop); // note subtle difference
00088         }
00089       }
00090     }else{                              /* no dash: a list                  */
00091       do {                              /* each arg                         */
00092         unsigned next = cmd.cursor();
00093         do {                            /* all that match this arg          */
00094           OPT::language->print_item(out, *ci);
00095           cmd.reset(arg1);
00096           assert(!ci.is_end());
00097           ci = findbranch(cmd, ++ci); // next match
00098         } while (!ci.is_end());
00099         cmd.reset(arg1 = next);
00100         ci = findbranch(cmd, scope); // next arg
00101       } while (!ci.is_end());
00102     }
00103   }
00104 }
