00139 void EVAL_BM_EXP::tr_eval(ELEMENT* d)const
00140 {
00141   double ev = _iv;
00142   for (double time = d->_sim->_time0;  time >= 0;  time -= _period) {
00143     if (time > _td1) {
00144       ev += (_pv - _iv) * (1. - exp(-(time-_td1)/_tau1));
00145     }else{
00146     }
00147     if (time > _td2) {
00148       ev += (_iv - _pv) * (1. - exp(-(time-_td2)/_tau2));
00149     }else{
00150     }
00151   }
00152   tr_finish_tdv(d, ev);
00153 }

00155 TIME_PAIR EVAL_BM_EXP::tr_review(COMPONENT* d)
00156 {
00157   double time = d->_sim->_time0;
00158   time += d->_sim->_dtmin * .01;  // hack to avoid duplicate events from numerical noise
00159   double raw_time = time;
00160 
00161   if (0 < _period && _period < BIGBIG) {
00162     time = fmod(time,_period);
00163   }else{
00164   }
00165   double time_offset = raw_time - time;
00166 
00167   double dt = NEVER;
00168   if (time > _td2) {
00169     d->_time_by.min_event(_period + time_offset);
00170     dt = (_tau2 > 0) ? _tau2 : NEVER;
00171   }else if (time > _td1) {
00172     d->_time_by.min_event(_td2 + time_offset);
00173     dt = (_tau1 > 0) ? _tau1 : NEVER;
00174   }else if (d->_sim->_time0 < _period) {
00175     d->_time_by.min_event(_td1 + time_offset);
00176     dt = NEVER;
00177   }else{
00178     d->_time_by.min_event(_td1 + time_offset);
00179     dt = (_tau2 > 0) ? _tau2 : NEVER;
00180   }
00181   d->_time_by.min_error_estimate(d->_sim->_time0 + dt);
00182 
00183   return d->_time_by;
00184 }
