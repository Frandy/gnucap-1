00087 double gen()
00088 {
00089   if (CKT_BASE::_sim->_time0 <= delay) {
00090     return init_;
00091   }
00092   double loctime = CKT_BASE::_sim->_time0 - delay;
00093   if (period > 0.) {
00094     untested();
00095     loctime = fmod(loctime, period);
00096   }
00097 
00098   double level;
00099   if (CKT_BASE::_sim->_time0 <= delay + rise) {                 /* initial rise */
00100     level = (maxv - 0) * (loctime/rise) + 0;
00101   }else if (loctime <= rise) {                          /* rising       */
00102     untested();
00103     level = (maxv - minv) * (loctime/rise) + minv;
00104   }else if (width==0.  ||  (loctime-=rise) <= width) {  /* pulse on     */
00105     level = maxv;
00106   }else if ((loctime-=width) <= fall) {                 /* falling      */
00107     untested();
00108     level = (minv - maxv) * (loctime/fall) + maxv;
00109   }else{                                                /* pulse off    */
00110     untested();
00111     level = minv;
00112   }
00113   level *= (freq == 0.) 
00114     ? ampl
00115     : ampl * sin(M_TWO_PI * freq*(CKT_BASE::_sim->_time0-delay) + phaz * DTOR);
00116   return (CKT_BASE::_sim->_time0 <= delay + rise)
00117     ? level + (offset - init_) * (loctime/rise) + init_
00118     : level + offset;
00119 }
