00032 void Expression::dump(std::ostream& out)const
00033 {
00034   std::vector<const Token*> locals; // a way of faking garbage collection.
00035   std::vector<const Token*> stack;  // actually use this
00036   // The _list is the expression in RPN.
00037   // Un-parse it -- back to infix.
00038   for (const_iterator i = begin(); i != end(); ++i) {
00039     if (dynamic_cast<const Token_STOP*>(*i)) {
00040       stack.push_back(*i);
00041     }else if (dynamic_cast<const Token_PARLIST*>(*i)) {
00042       // pop*n  push
00043       bool been_here = false;
00044       std::string tmp(")");
00045       for (;;) {
00046         if (stack.empty()) {untested();
00047           throw Exception("bad expression");
00048         }else{
00049         }
00050         const Token* t = stack.back();
00051         stack.pop_back();
00052         if (dynamic_cast<const Token_STOP*>(t)) {
00053           tmp = "(" + tmp;
00054           break;
00055         }else if (dynamic_cast<const Token_SYMBOL*>(t)) {
00056           if (been_here) {
00057             tmp = ", " + tmp;
00058           }else{
00059             been_here = true;
00060           }
00061           tmp = t->full_name() + tmp;
00062         }else{
00063           unreachable();
00064         }
00065       }
00066       Token* t = new Token_PARLIST(tmp);
00067       locals.push_back(t);
00068       stack.push_back(t);
00069     }else if (dynamic_cast<const Token_CONSTANT*>(*i)|| dynamic_cast<const Token_SYMBOL*>(*i)) {
00070       if (!stack.empty() && (dynamic_cast<const Token_PARLIST*>(stack.back()))) {
00071         // has parameters (table or function)
00072         // pop op push
00073         const Token* t1 = stack.back();
00074         stack.pop_back();
00075         Token* t = new Token_SYMBOL((**i).name(), t1->full_name());
00076         locals.push_back(t);
00077         stack.push_back(t);
00078       }else{
00079         // has no parameters (scalar)
00080         stack.push_back(*i);
00081       }
00082     }else if (dynamic_cast<const Token_BINOP*>(*i)) {
00083       // pop pop op push
00084       assert(!stack.empty());
00085       const Token* t2 = stack.back();
00086       stack.pop_back();
00087       assert(!stack.empty());
00088       const Token* t1 = stack.back();
00089       stack.pop_back();
00090       std::string tmp('(' + t1->full_name() + ' ' + (**i).name() + ' ' + t2->full_name() + ')');
00091       Token* t = new Token_SYMBOL(tmp, "");
00092       locals.push_back(t);
00093       stack.push_back(t);
00094     }else if (dynamic_cast<const Token_UNARY*>(*i)) {
00095       // pop op push
00096       assert(!stack.empty());
00097       const Token* t1 = stack.back();
00098       stack.pop_back();
00099       std::string tmp('(' + (**i).name() + ' ' + t1->full_name() + ')');
00100       Token* t = new Token_SYMBOL(tmp, "");
00101       locals.push_back(t);
00102       stack.push_back(t);
00103     }else{
00104       unreachable();
00105     }
00106   }
00107   if (stack.empty()) {untested();
00108     out << "empty";
00109   }else{
00110     out << stack.back()->full_name();
00111     assert(stack.size() == 1);
00112   }
00113   while (!locals.empty()) {
00114     delete locals.back();
00115     locals.pop_back();
00116   }
00117 }
