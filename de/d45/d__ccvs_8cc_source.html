00086 bool DEV_CCVS::do_tr_last()
00087 {
00088   assert(_input);
00089   if (using_tr_eval()) {

00097     assert(_y[0].f1 == value());
00098     _m0.c0 = 0.;
00099     assert(converged());
00100   }
00101   if (_input->has_inode()) {untested();
00102     // nothing
00103   }else if (_input->has_iv_probe()) {
00104     _m0.c0 += _y[0].f1 * _input->_m0.c0;
00105     _m0.c1  = _y[0].f1 * (_input->_loss0 + _input->_m0.c1);
00106   }else{unreachable();
00107   }
00108   _m0 *= -_loss0;
00109   store_values();
00110   q_load();
