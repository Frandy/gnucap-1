00388 double ELEMENT::tr_review_trunc_error(const FPOLY1* q)
00389 {
00390   int error_deriv = order()+1;
00391   double timestep;
00392   if (_time[0] <= 0.) {
00393     // DC, I know nothing
00394     timestep = NEVER;
00395   }else if (_time[error_deriv] <= 0.) {
00396     // first few steps, I still know nothing
00397     // repeat whatever step was used the first time
00398     timestep = _dt;
00399   }else{
00400     for (int i=error_deriv; i>0; --i) {
00401       assert(_time[i] < _time[i-1]); // || _time[i] == 0.);
00402     }
00403 
00404     double c[OPT::_keep_time_steps];
00405     for (int i=0; i<OPT::_keep_time_steps; ++i) {
00406       c[i] = q[i].f0;
00407     }
00408     assert(error_deriv < OPT::_keep_time_steps);
00409     derivatives(c, OPT::_keep_time_steps, _time);
00410     // now c[i] is i'th derivative
00411     
00412     assert(OPT::_keep_time_steps >= 5);
00413     trace0(("ts" + long_label()).c_str());
00414     trace5("time", _time[0], _time[1], _time[2], _time[3], _time[4]);
00415     trace5("charge", q[0].f0, q[1].f0, q[2].f0, q[3].f0, q[4].f0);
00416     trace5("deriv", c[0], c[1], c[2], c[3], c[4]);
00417     
00418     if (c[error_deriv] == 0) {
00419       timestep = NEVER;
00420     }else{
00421       double chargetol = std::max(OPT::chgtol,
00422         OPT::reltol * std::max(std::abs(q[0].f0), std::abs(q[1].f0)));
00423       double tol = OPT::trtol * chargetol;
00424       double denom = error_factor() * std::abs(c[error_deriv]);
00425       assert(tol > 0.);
00426       assert(denom > 0.);
00427       switch (error_deriv) { // pow is slow.
00428       case 1:   timestep = tol / denom;         break;
00429       case 2:   timestep = sqrt(tol / denom);   break;
00430       case 3:   timestep = cbrt(tol / denom);   break;
00431       default:  timestep = pow((tol / denom), 1./(error_deriv)); break;
00432       }
00433       trace4("", chargetol, tol, denom, timestep);
00434     }
00435   }
00436   assert(timestep > 0.);
00437   return timestep;
00438 }
