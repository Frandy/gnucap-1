00039     // RTLD_NOW means to resolve symbols on loading
00040     // RTLD_LOCAL means symbols defined in a plugin are local
00041     do {
00042       if (cmd.umatch("public ")) {
00043         dl_scope = RTLD_GLOBAL;
00044         // RTLD_GLOBAL means symbols defined in a plugin are global
00045         // Use this when a plugin depends on another.
00046       }else if (cmd.umatch("lazy ")) {
00047         check = RTLD_LAZY;
00048         // RTLD_LAZY means to defer resolving symbols until needed
00049         // Use when a plugin will not load because of unresolved symbols,
00050         // but it may work without it.
00051       }else{
00052       }
00053     } while (cmd.more() && !cmd.stuck(&here));
00054 
00055     std::string file_name;
00056     cmd >> file_name;
00057     
00058     void* handle = attach_list[file_name];
00059     if (handle) {
00060       if (CARD_LIST::card_list.is_empty()) {
00061         cmd.warn(bDANGER, here, "\"" + file_name + "\": already loaded, replacing");
00062         dlclose(handle);

00064       }else{untested();

00066         throw Exception_CS("already loaded, cannot replace when there is a circuit", cmd);
00067       }
00068     }else{
00069     }
