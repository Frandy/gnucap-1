00524 void DEV_TRANSLINE::do_ac()
00525 {
00526   const COMMON_TRANSLINE*c=prechecked_cast<const COMMON_TRANSLINE*>(common());
00527   assert(c);
00528   double lenth = _sim->_freq * c->real_td * 4;  /* length in quarter waves */
00529   double dif = lenth - floor(lenth+.5); /* avoid divide by zero if close to */
00530   if (std::abs(dif) < LINLENTOL) {      /* resonance by tweeking a little */
00531     error(bDEBUG,
00532           long_label() + ": transmission line too close to resonance\n");
00533     lenth = (dif<0.) ? floor(lenth+.5)-LINLENTOL : floor(lenth+.5)+LINLENTOL;
00534   }else{
00535   }
00536   lenth *= (M_PI_2);    /* now in radians */
00537   
00538   _y12 = COMPLEX(0., -1. / (c->real_z0 * sin(lenth)));
00539   _y11 = COMPLEX(0., tan(lenth/2) / c->real_z0) + _y12;
00540 }
