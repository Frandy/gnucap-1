00060  * arg is name of file

00067   void do_it(CS& cmd, CARD_LIST*)
00068   {itested();

00070     
00071     if (cmd.more()) {                   /* a file name .. open it */
00072       const char *access = "w";
00073       while (cmd.match1('>')) {
00074         access = "a";
00075         cmd.skip();
00076         cmd.skipbl();
00077       }
00078       FILE* newfile = xopen(cmd,"",access);
00079       if (newfile) {
00080         filestack.push_back(newfile);
00081         mlog.attach(newfile);
00082       }else{
00083       }
00084     }else{                              /* empty command -- close a file */
00085       if (filestack.empty()) {
00086         error(bWARNING, "no files open\n");
00087       }else{
00088         FILE* oldfile = filestack.back();
