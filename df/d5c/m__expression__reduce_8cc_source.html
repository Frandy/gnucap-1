00115 {
00116   assert(E);
00117   // replace single token with its value
00118   if (!E->is_empty() && dynamic_cast<const Token_PARLIST*>(E->back())) {
00119     // has parameters (table or function)
00120     if (FUNCTION* f = function_dispatcher[name()]) {
00121       const Token* T1 = E->back(); // arglist
00122       E->pop_back();
00123       CS cmd(CS::_STRING, T1->name());      
00124       std::string value = f->eval(cmd, E->_scope);
00125       const Float* v = new Float(value);
00126       E->push_back(new Token_CONSTANT(value, v, ""));
00127       delete T1;
00128     }else{
00129       throw Exception_No_Match(name()); //BUG// memory leak
00130       unreachable();
00131       E->push_back(clone());
00132     }
00133   }else{
00134     // has no parameters (scalar)
00135     if (strchr("0123456789.", name()[0])) {
00136       // a number
00137       Float* n = new Float(name());
00138       E->push_back(new Token_CONSTANT(name(), n, ""));
00139     }else{
00140       // a name
00141       PARAMETER<double> p = (*(E->_scope->params()))[name()];
00142       if (p.has_hard_value()) {
00143         // can find value - push value
00144         double v = p.e_val(NOT_INPUT, E->_scope);
00145         Float* n = new Float(v);
00146         E->push_back(new Token_CONSTANT(n->val_string(), n, ""));
00147       }else{
00148         // no value - push name (and accept incomplete solution later)
00149         String* s = new String(name());
00150         E->push_back(new Token_CONSTANT(name(), s, ""));        
00151       }
00152     }
